fn kmph_to_ms(kmph: f64) -> f64 {
    let meters_per_hour = kmph * 1000.0;
    let seconds_per_hour = 60.0 * 60.0;
    meters_per_hour / seconds_per_hour
}

fn ms_to_kmph(ms: f64) -> f64 {
    let seconds_per_hour = 60.0 * 60.0;
    let km_per_second = ms / 1000.0;
    km_per_second * seconds_per_hour
}

fn main() {
    println!("{}", kmph_to_ms(18.0));
    println!("{}", ms_to_kmph(5.0));
}
